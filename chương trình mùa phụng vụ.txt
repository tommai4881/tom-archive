void main()
{
    int n,tt;
    do
    {
        printf("=======================================\n");
        printf("1. Mùa Vọng\n");
        printf("2. Mùa Giáng Sinh\n");
        printf("3. Mùa Tết Nguyên Đán\n");
        printf("4. Mùa Chay\n");
        printf("5. Mùa Phục Sinh và Phật Đản\n");
        printf("6. Mùa Hiện Xuống\n");
        printf("7. Mùa Vu Lan, Quốc Khánh và Trung Thu\n");
        printf("8. Mùa Mân Côi và Nước Trời\n");
        printf("Nhập tên mùa: ");
        scanf("%d",&n);
        switch(n)
        {
            case 1: advent(); break;
            case 2: christmastide(); break;
            case 3: newyear(); break;
            case 4: lent(); break;
            case 5: eastertide(); break;
            case 6: pentecost(); break;
            case 7: ullambana(); break;
            case 8: rosary(); break;
            default: printf("Mùa vừa nhập không hợp lệ!");
        };
        printf("\nNhấn số 1 để thử lại! ");
        scanf("%d",&tt);
    } while (tt==1);
    printf("=======================================\n");
    printf("Chương trình kết thúc.\n");
    printf("=======================================\n");
}
